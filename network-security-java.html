Network Security trong Java: Bảo mật ứng dụng mạng

Java 12/12/2024 4 phút đọc

Bảo mật mạng là một khía cạnh quan trọng trong phát triển ứng dụng Java.

1. Nó là gì?

SSL/TLS: Giao thức mã hóa để bảo mật kết nối mạng giữa client và server.

Authentication: Xác thực danh tính người dùng hoặc hệ thống.

Authorization: Kiểm soát quyền truy cập vào tài nguyên.

Encryption: Mã hóa dữ liệu để bảo vệ thông tin nhạy cảm.

2. Nó giúp ích cho tôi ra sao khi sử dụng?

  • Bảo vệ dữ liệu: Mã hóa thông tin nhạy cảm trong quá trình truyền tải
  • Xác thực người dùng: Đảm bảo chỉ người dùng hợp lệ mới có thể truy cập
  • Kiểm soát quyền: Phân quyền truy cập theo vai trò và chức năng
  • Tuân thủ quy định: Đáp ứng các yêu cầu bảo mật của pháp luật
  • Xây dựng niềm tin: Tăng độ tin cậy của người dùng với ứng dụng

3. Cách sử dụng

3.1. SSL/TLS với Spring Boot

// application.yml
server:
  port: 8443
  ssl:
    enabled: true
    key-store: classpath:keystore.p12
    key-store-password: password
    key-store-type: PKCS12
    key-alias: tomcat

// SSL Configuration
@Configuration
public class SSLConfig {
    
    @Bean
    public TomcatServletWebServerFactory tomcatServletWebServerFactory() {
        TomcatServletWebServerFactory factory = new TomcatServletWebServerFactory();
        factory.addConnectorCustomizers(connector -> {
            connector.setSecure(true);
            connector.setScheme("https");
            connector.setPort(8443);
        });
        return factory;
    }
}

3.2. JWT Authentication

// JWT Service
@Service
public class JwtService {
    
    private final String SECRET_KEY = "mySecretKey";
    
    public String generateToken(String username) {
        return Jwts.builder()
            .setSubject(username)
            .setIssuedAt(new Date())
            .setExpiration(new Date(System.currentTimeMillis() + 86400000)) // 24h
            .signWith(SignatureAlgorithm.HS256, SECRET_KEY)
            .compact();
    }
    
    public boolean validateToken(String token) {
        try {
            Jwts.parser().setSigningKey(SECRET_KEY).parseClaimsJws(token);
            return true;
        } catch (JwtException e) {
            return false;
        }
    }
}

// Security Configuration
@Configuration
@EnableWebSecurity
public class SecurityConfig {
    
    @Bean
    public SecurityFilterChain filterChain(HttpSecurity http) throws Exception {
        http.csrf().disable()
            .authorizeHttpRequests(authz -> authz
                .requestMatchers("/api/public/**").permitAll()
                .requestMatchers("/api/admin/**").hasRole("ADMIN")
                .anyRequest().authenticated()
            )
            .addFilterBefore(jwtAuthenticationFilter(), UsernamePasswordAuthenticationFilter.class);
        
        return http.build();
    }
}

3.3. Input Validation và Sanitization

// Input Validation
@RestController
public class UserController {
    
    @PostMapping("/api/users")
    public ResponseEntity createUser(@Valid @RequestBody CreateUserRequest request) {
        // Validate input
        if (request.getEmail() == null || !isValidEmail(request.getEmail())) {
            throw new ValidationException("Invalid email format");
        }
        
        // Sanitize input
        String sanitizedName = sanitizeInput(request.getName());
        
        User user = new User();
        user.setName(sanitizedName);
        user.setEmail(request.getEmail());
        
        return ResponseEntity.ok(userService.save(user));
    }
    
    private boolean isValidEmail(String email) {
        return email.matches("^[A-Za-z0-9+_.-]+@(.+)$");
    }
    
    private String sanitizeInput(String input) {
        return input.replaceAll("[<>\"'&]", "");
    }
}
